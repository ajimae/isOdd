## Usage

```typescript
import { isOdd } from "https://raw.githubusercontent.com/ajimae/isOdd/master/mod.ts";

var numberOne = 3;
const checkOne = isOdd(numberOne);
console.log(checkOne); // true

var numberTwo = 8;
const checkTwo = isOdd(numberTwo);
console.log(checkOne); // false
```
